PLATAFORMA DE MONITOREO — PERÚ

1. Configura Supabase (gratis, sin tarjeta)
→ Ve a supabase.com → New project → copia la URL y la anon key de Settings → API
2. Ejecuta este SQL en Supabase → SQL Editor
-- FleetTrack GPS — Tablas (copiar y ejecutar en Supabase SQL Editor) create table if not exists ft_users ( id uuid default gen_random_uuid() primary key, name text not null, email text unique not null, password text not null, role text default 'driver', unit_id text, color text default '#3d7fff', created_at timestamptz default now() ); create table if not exists ft_fleet ( unit_id text primary key, driver_name text, route_name text, origin_name text, dest_name text, lat float8, lng float8, speed float8 default 0, heading float8 default 0, accuracy float8, dist float8 default 0, status text default 'offline', updated_at timestamptz default now() ); create table if not exists ft_trails ( id uuid default gen_random_uuid() primary key, unit_id text, lat float8, lng float8, speed float8 default 0, ts timestamptz default now() ); create table if not exists ft_routes ( id uuid default gen_random_uuid() primary key, name text not null, origin_name text, dest_name text, created_at timestamptz default now() ); alter table ft_users enable row level security; alter table ft_fleet enable row level security; alter table ft_trails enable row level security; alter table ft_routes enable row level security; create policy "open" on ft_users for all to anon using (true) with check (true); create policy "open" on ft_fleet for all to anon using (true) with check (true); create policy "open" on ft_trails for all to anon using (true) with check (true); create policy "open" on ft_routes for all to anon using (true) with check (true); alter publication supabase_realtime add table ft_fleet; -- Admin por defecto (cambiar contraseña después) insert into ft_users (name,email,password,role) values ('Administrador','[email protected]','admin123','admin') on conflict (email) do nothing;
3. Crea tu cuenta de administrador
